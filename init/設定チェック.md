```bash

```

```bash
#! /bin/sh


# Cドライブのマウント
sudo umount C_DRIVE
sudo mount -t vboxsf C_DRIVE /mnt/c/
df -h


# Dドライブのマウント <- Dドライブ（外付けSSD）をマウントした場合
sudo umount D_DRIVE
sudo mount -t vboxsf D_DRIVE /mnt/d/
df -h


# PostgreSQLサーバの起動
sudo service postgresql start
```
